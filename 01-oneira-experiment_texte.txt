(function initCanvas() {
  const canvas = document.getElementById('bg-canvas');
  const ctx = canvas.getContext('2d');
  let W, H, stars = [], nebula = [];

  function resize() {
    W = canvas.width = window.innerWidth;
    H = canvas.height = window.innerHeight;
    stars = [];
    for (let i = 0; i < 120; i++) {
      stars.push({
        x: Math.random() * W,
        y: Math.random() * H,
        r: Math.random() * 1.2,
        a: Math.random(),
        speed: 0.0005 + Math.random() * 0.001,
        phase: Math.random() * Math.PI * 2
      });
    }
    nebula = [];
    for (let i = 0; i < 4; i++) {
      nebula.push({
        x: Math.random() * W,
        y: Math.random() * H,
        r: 200 + Math.random() * 300,
        color: i % 2 === 0 ? 'rgba(109,40,217,' : 'rgba(30,58,138,',
        a: 0.02 + Math.random() * 0.04
      });
    }
  }

  let t = 0;
  function draw() {
    t += 0.003;
    ctx.clearRect(0, 0, W, H);

    nebula.forEach(n => {
      const grd = ctx.createRadialGradient(n.x, n.y, 0, n.x, n.y, n.r);
      grd.addColorStop(0, n.color + n.a + ')');
      grd.addColorStop(1, n.color + '0)');
      ctx.fillStyle = grd;
      ctx.beginPath();
      ctx.arc(n.x, n.y, n.r, 0, Math.PI * 2);
      ctx.fill();
    });

    stars.forEach(s => {
      const flicker = s.a * (0.5 + 0.5 * Math.sin(t * s.speed * 1000 + s.phase));
      ctx.globalAlpha = flicker;
      ctx.fillStyle = '#F9F8F5';
      ctx.beginPath();
      ctx.arc(s.x, s.y, s.r, 0, Math.PI * 2);
      ctx.fill();
    });
    ctx.globalAlpha = 1;
    requestAnimationFrame(draw);
  }

  window.addEventListener('resize', resize);
  resize();
  draw();
})();

function startAudio() {
  if (S.audioCtx) return;
  try {
    S.audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    const ctx = S.audioCtx;
    const master = ctx.createGain();
    master.gain.setValueAtTime(0, ctx.currentTime);
    master.gain.linearRampToValueAtTime(0.06, ctx.currentTime + 4);
    master.connect(ctx.destination);

    // Deep drone
    [40, 80, 120].forEach((freq, i) => {
      const osc = ctx.createOscillator();
      const gain = ctx.createGain();
      osc.type = 'sine';
      osc.frequency.setValueAtTime(freq + i * 0.15, ctx.currentTime);
      gain.gain.setValueAtTime(1 / (i + 1), ctx.currentTime);
      osc.connect(gain);
      gain.connect(master);
      osc.start();
      S.audioNodes.push(osc);
    });

    // Subtle high shimmer
    const shim = ctx.createOscillator();
    const shimGain = ctx.createGain();
    shimGain.gain.setValueAtTime(0.02, ctx.currentTime);
    shim.type = 'sine';
    shim.frequency.setValueAtTime(528, ctx.currentTime);
    shim.connect(shimGain);
    shimGain.connect(master);
    shim.start();
    S.audioNodes.push(shim);

    S.audioNodes.push(master);
  } catch(e) { console.log('Audio not available'); }
}

function stopAudio() {
  if (!S.audioCtx) return;
  try {
    S.audioNodes.forEach(n => { try { n.stop ? n.stop() : n.disconnect(); } catch(e) {} });
    S.audioCtx.close();
  } catch(e) {}
  S.audioCtx = null;
  S.audioNodes = [];
}

const STOP_WORDS = new Set(['je','tu','il','elle','nous','vous','ils','elles','le','la','les','un','de','du','des','et','en','au','aux','à','ce','qui','que','quoi','mon','ma','mes','son','sa','ses','sur','dans','par','pour','avec','est','sont','était','ont','avoir','être','pas','plus','très','mais','ou','si','ne','se','me','te','lui','leur','y','bien','tout','aussi','même','cette','cet','ces','quel','quelle','dont','où','car','ni','donc','or','quand','comme','puis','alors','encore','déjà','toujours','jamais','souvent','peu','beaucoup']);

function extractKeywords() {
  const allText = [];
  for (let n = 1; n <= 7; n++) {
    ['notes','q1','q2','q3','zones','enjeu','recurrences','surprise','t1','t2','t3','pont','images','emotions','perso','actions','atmo'].forEach(f => {
      const el = document.getElementById('n' + n + '-' + f);
      if (el?.value) allText.push(el.value);
    });
  }

  const words = allText.join(' ')
    .toLowerCase()
    .replace(/[.,!?;:'"«»\(\)\[\]]/g, ' ')
    .split(/\s+/)
    .filter(w => w.length > 3 && !STOP_WORDS.has(w));

  const freq = {};
  words.forEach(w => { freq[w] = (freq[w] || 0) + 1; });
  return Object.entries(freq)
    .filter(([,v]) => v >= 2)
    .sort((a,b) => b[1] - a[1])
    .slice(0, 8)
    .map(([w]) => w);
}

function getSymbolCounts() {

    (arr || []).forEach(tag => { counts[tag] = (counts[tag] || 0) + 1; });
  });
  return counts;
}

function getDominantSymbol() {
  const counts = getSymbolCounts();
  const entries = Object.entries(counts);
  if (!entries.length) return null;
  return entries.sort((a,b) => b[1]-a[1])[0];
}

function getAvgScore() {
  const scores = Object.values(S.scores);
  if (!scores.length) return null;
  return (scores.reduce((a,b)=>a+b,0) / scores.length);
}

// MODULE 1 — DREAM CONSTELLATION

let constellationRAF = null;

function buildConstellationNodes() {
  const counts = getSymbolCounts();
  const entries = Object.entries(counts).filter(([,v]) => v > 0);
  if (!entries.length) return [];

  const maxCount = Math.max(...entries.map(([,v]) => v));
  const nodes = entries.map(([sym, count], i) => {
    const angle = (i / entries.length) * Math.PI * 2 - Math.PI / 2;
    const radius = 0.28 + (1 - count / maxCount) * 0.1;
    return {
      sym,
      count,
      r: 8 + (count / maxCount) * 22,
      nx: 0.5 + Math.cos(angle) * radius,
      ny: 0.5 + Math.sin(angle) * radius,
      phase: Math.random() * Math.PI * 2,
      speed: 0.3 + Math.random() * 0.5,
    };
  });

  // Also find co-occurrences (same night)
  const coEdges = [];
  for (let n = 1; n <= 7; n++) {
    const nightTags = S.tags['n' + n] || [];
    for (let i = 0; i < nightTags.length; i++) {
      for (let j = i + 1; j < nightTags.length; j++) {
        const a = nodes.findIndex(n => n.sym === nightTags[i]);
        const b = nodes.findIndex(n => n.sym === nightTags[j]);
        if (a !== -1 && b !== -1) {
          const existing = coEdges.find(e =>
            (e.a === a && e.b === b) || (e.a === b && e.b === a));
          if (existing) existing.w++;
          else coEdges.push({ a, b, w: 1 });
        }
      }
    }
  }

  return { nodes, edges: coEdges };
}

function drawConstellation(canvasId, compact = false) {
  const canvas = document.getElementById(canvasId);
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  const W = canvas.width;
  const H = canvas.height;
  const data = buildConstellationNodes();

  if (!data.nodes || !data.nodes.length) {
    ctx.clearRect(0, 0, W, H);
    ctx.fillStyle = 'rgba(249,248,245,0.15)';
    ctx.font = '11px DM Mono, monospace';
    ctx.textAlign = 'center';
    ctx.fillText('Sélectionnez des symboles dans les nuits pour les voir apparaître', W/2, H/2);
    return;
  }

  const { nodes, edges } = data;
  let t = 0;

  function frame() {
    t += 0.008;
    ctx.clearRect(0, 0, W, H);

    // Background glow
    const bg = ctx.createRadialGradient(W/2, H/2, 0, W/2, H/2, W * 0.6);
    bg.addColorStop(0, 'rgba(109,40,217,0.06)');
    bg.addColorStop(1, 'rgba(8,14,24,0)');
    ctx.fillStyle = bg;
    ctx.fillRect(0, 0, W, H);

    // Draw edges
    edges.forEach(edge => {
      const na = nodes[edge.a];
      const nb = nodes[edge.b];
      const ax = na.nx * W;
      const ay = na.ny * H;
      const bx = nb.nx * W;
      const by = nb.ny * H;
      const alpha = Math.min(0.4, edge.w * 0.15);
      ctx.beginPath();
      ctx.moveTo(ax, ay);
      ctx.lineTo(bx, by);
      ctx.strokeStyle = `rgba(139,92,246,${alpha})`;
      ctx.lineWidth = edge.w * 0.5;
      ctx.stroke();
    });

    // Draw nodes
    nodes.forEach(node => {
      const x = node.nx * W;
      const y = node.ny * H;
      const pulse = 1 + 0.06 * Math.sin(t * node.speed + node.phase);
      const r = node.r * pulse;

      // Glow
      const glow = ctx.createRadialGradient(x, y, 0, x, y, r * 3.5);
      glow.addColorStop(0, `rgba(139,92,246,${0.2 * (node.count / 7)})`);
      glow.addColorStop(1, 'rgba(139,92,246,0)');
      ctx.fillStyle = glow;
      ctx.beginPath();
      ctx.arc(x, y, r * 3.5, 0, Math.PI * 2);
      ctx.fill();

      // Node body
      const grad = ctx.createRadialGradient(x - r*0.3, y - r*0.3, 0, x, y, r);
      grad.addColorStop(0, 'rgba(249,248,245,0.95)');
      grad.addColorStop(0.4, 'rgba(139,92,246,0.8)');
      grad.addColorStop(1, 'rgba(109,40,217,0.4)');
      ctx.fillStyle = grad;
      ctx.beginPath();
      ctx.arc(x, y, r, 0, Math.PI * 2);
      ctx.fill();

      // Label
      if (!compact) {
        ctx.fillStyle = 'rgba(249,248,245,0.85)';
        ctx.font = `${Math.max(9, 9 + node.count)}px DM Mono, monospace`;
        ctx.textAlign = 'center';
        ctx.fillText(node.sym, x, y + r + 14);

        // Count badge
        if (node.count > 1) {
          ctx.fillStyle = 'rgba(139,92,246,0.7)';
          ctx.font = '8px DM Mono, monospace';
          ctx.fillText('×' + node.count, x, y + r + 24);
        }
      } else {
        ctx.fillStyle = 'rgba(249,248,245,0.6)';
        ctx.font = '8px DM Mono, monospace';
        ctx.textAlign = 'center';
        ctx.fillText(node.sym, x, y + r + 11);
      }
    });

    constellationRAF = requestAnimationFrame(frame);
  }

  if (constellationRAF) cancelAnimationFrame(constellationRAF);
  frame();
}

function openConstellation() {
  const nightsDone = S.completed.length;
  if (nightsDone < 5) {

  }
  renderPatternEngine('constellation-pattern');

  setTimeout(() => drawConstellation('constellation-canvas', false), 400);
}

    conData.nodes.forEach(node => {
      const nx = cX + (node.nx - 0.5) * cR * 2;
      const ny = cY + (node.ny - 0.5) * cR * 2;
      const r = 4 + node.count * 3;
      const grad = ctx.createRadialGradient(nx, ny, 0, nx, ny, r);
      grad.addColorStop(0, 'rgba(249,248,245,0.9)');
      grad.addColorStop(1, 'rgba(139,92,246,0.5)');
      ctx.fillStyle = grad;
      ctx.beginPath();
      ctx.arc(nx, ny, r, 0, Math.PI*2);
      ctx.fill();
    });
  }

  // Nights completed
  const nights = S.completed.length;
  for (let i = 0; i < 7; i++) {
    const filled = i < nights;
    ctx.fillStyle = filled ? 'rgba(139,92,246,0.7)' : 'rgba(249,248,245,0.12)';
    ctx.beginPath();
    ctx.arc(48 + i * 20, H - 55, 5, 0, Math.PI*2);
    ctx.fill();
  }

  ctx.fillStyle = 'rgba(249,248,245,0.2)';
  ctx.font = '8px DM Mono, monospace';
  ctx.textAlign = 'left';
  ctx.fillText(nights + '/7 nuits · ' + new Date().toLocaleDateString('fr-FR'), 48, H - 35);

  // oneira.io watermark
  ctx.fillStyle = 'rgba(139,92,246,0.4)';
  ctx.font = '8px DM Mono, monospace';
  ctx.textAlign = 'right';
  ctx.fillText('ONEIRA · Le Laboratoire du Sommeil', W - 48, H - 35);

  document.getElementById('share-download-btn').style.display = 'inline-block';
  toast('Carte générée ✦');
}

function updateConstellationTeaser() {
  const teaser = document.getElementById('constellation-teaser');
  if (!teaser) return;

  const allTags = Object.values(S.tags).flat();
  const hasTags = allTags.length > 0;

  if (hasTags || S.completed.length >= 1) {
    teaser.style.display = 'block';
  }

  const lockMsg = document.getElementById('teaser-lock-msg');
  if (S.completed.length >= 5) {
    if (lockMsg) lockMsg.textContent = '→ Ouvrir la constellation';

    teaser.style.opacity = '0.6';

}

function saveAll() {
  const data = {
    question: document.getElementById('ob-question')?.value || '',
    scores: S.scores,
    tags: S.tags,
    completed: S.completed,
    notes: {}
  };
  document.querySelectorAll('textarea[id], input[id]').forEach(el => {
    data.notes[el.id] = el.value;
  });
  try { localStorage.setItem('oneira-v2', JSON.stringify(data)); } catch(e) {}
  updateChart();
  updateSymbolMap();
}

function loadAll() {
  try {
    const raw = localStorage.getItem('oneira-v2');
    if (!raw) return;
    const data = JSON.parse(raw);

    if (data.question) {
      const qEl = document.getElementById('ob-question');
      if (qEl) qEl.value = data.question;
      document.getElementById('week-q-display').textContent = data.question;
    }
    if (data.scores) Object.assign(S.scores, data.scores);
    if (data.tags) Object.assign(S.tags, data.tags);
    if (data.completed) S.completed = data.completed;

    if (data.notes) {
      Object.keys(data.notes).forEach(id => {
        const el = document.getElementById(id);
        if (el) el.value = data.notes[id];
      });
    }

    // Restore scores UI
    Object.keys(S.scores).forEach(n => {
      const v = S.scores[n];
      for (let i = 1; i <= 5; i++) {
        const b = document.getElementById('n' + n + '-s' + i);
        if (b) b.classList.toggle('sel', i === v);
      }
    });

    // Restore completed
    S.completed.forEach(n => {
      const btn = document.getElementById('n' + n + '-done');
      if (btn) { btn.classList.add('done'); btn.textContent = '✦   Nuit ' + n + ' complète'; }
      const pill = document.getElementById('pill-' + n);
      if (pill) { pill.classList.remove('active'); pill.classList.add('done'); }
    });

    // Restore tags
    Object.keys(S.tags).forEach(night => {
      const selectedTags = S.tags[night];
      const container = document.getElementById('tags-' + night);
      if (container) {
        container.querySelectorAll('.sym-tag').forEach(btn => {
          if (selectedTags.includes(btn.textContent)) btn.classList.add('selected');
        });
      }
    });

    updateChart();
    updateSymbolMap();

    if (data.question) goTo('app');
  } catch(e) {}
}

  const q = document.getElementById('ob-question')?.value || '—';

function importJSON(input) {
  const file = input.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = function(e) {
    try {
      const data = JSON.parse(e.target.result);
      // Restore question
      if (data.question) {
        const q = document.getElementById('ob-question');
        if (q) q.value = data.question;
      }
      // Restore scores
      if (data.scores) {
        Object.entries(data.scores).forEach(([night, score]) => {
          S.scores[night] = score;
          setScore(parseInt(night), score, true);
        });
      }
      // Restore tags
      if (data.tags) {
        S.tags = data.tags;
      }
      // Restore completed nights
      if (data.completed) {
        S.completed = data.completed;
      }
      // Restore all textarea values
      if (data.notes) {
        Object.entries(data.notes).forEach(([id, value]) => {
          const el = document.getElementById(id);
          if (el) el.value = value;
        });
      }
      saveAll();
      // Reset file input so same file can be re-imported
      input.value = '';
      toast('Données restaurées ✓');
    } catch(err) {
      toast('Fichier JSON invalide');
      input.value = '';
    }
  };
  reader.readAsText(file);
}

function sendData() {
  const emailEl = document.getElementById('capture-email');
  const email = emailEl ? emailEl.value.trim() : '';

  if (!email || !email.includes('@')) {
    toast('Entrez une adresse email valide');
    return;
  }

  const btn = document.getElementById('send-btn');
  btn.textContent = '…';
  btn.style.opacity = '0.5';

  // Build data summary
  const q = document.getElementById('ob-question')?.value || '—';
  let dataStr = '';
  for (let n = 1; n <= 7; n++) {
    if (!S.completed.includes(n)) continue;
    dataStr += `\nNuit ${n}`;
    if (S.scores[n]) dataStr += ` · Recall ${S.scores[n]}/5`;
    if (S.tags[n]?.length) dataStr += ` · Symboles: ${S.tags[n].join(', ')}`;
    const noteEl = document.getElementById('n' + n + '-notes');
    if (noteEl?.value) dataStr += `\nNotes: ${noteEl.value}`;
    ['q1','q2','q3'].forEach(q => {
      const el = document.getElementById('n' + n + '-' + q);
      if (el?.value) dataStr += `\n→ ${el.value}`;
    });
    dataStr += '\n';
  }

  const params = {
    user_email: email,
    name: email,
    email: email,
    question: q,
    data: dataStr || '(aucune nuit complétée)',
    date: new Date().toLocaleDateString('fr-CH', {
      day: '2-digit', month: 'long', year: 'numeric'
    })
  };

  fetch('https://formspree.io/f/mzdyzbnv', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
    body: JSON.stringify(params),
  })
    .then(() => {
      document.getElementById('capture-confirm').style.display = 'block';
      document.getElementById('capture-error').style.display = 'none';
      emailEl.style.display = 'none';
      btn.style.display = 'none';
    })
    .catch(() => {
      document.getElementById('capture-error').style.display = 'block';
      btn.textContent = 'Envoyer →';
      btn.style.opacity = '1';
    });
}
